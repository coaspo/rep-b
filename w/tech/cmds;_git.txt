1. check in (typical use):
       git add *
       git commit -m "message.."
       git push origin {branch}
2. merge:
       git checkout master
       git merge {branch}
       git push origin master
       branch -d  {ranch}  # delete branch
3. squash:
       git checkout master
       git merge {branch}
       git reset origin/master
       git add . --all
       git push
4. update local:
       git fetch origin {branch}
       git pull
